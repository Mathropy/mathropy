<script type="text/tikz">
  \begin{tikzpicture}[scale=2, line join=round]
    \pgfmathsetmacro{\sq}{sqrt(3)}
    \pgfmathsetmacro{\x}{2/sqrt(3) - 1}
    \pgfmathsetmacro{\h}{2 - sqrt(3)}
    \tikzset{
      rect/.style={draw=red, thick},
      main/.style={draw=black, thick}
    }

    \draw[main, dashed, gray] (0,0) rectangle (2,2);
    \draw[main, blue] (0,0) -- (2,0) -- (1,\sq) -- cycle;

    \draw[rect] (0,\sq) rectangle (1,2);
    
    \foreach \i in {0, 1, 2, 3} {
      \draw[rect, fill=orange, fill opacity=0.2] 
        ({\x + \i*\x}, {\i*\h}) rectangle +(1, \h);
    }
  \end{tikzpicture}
</script>
